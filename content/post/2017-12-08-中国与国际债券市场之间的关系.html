用R对数据进行初步处理

数据是来自彭博巴克莱指数系列的各国国债指数，节假日没有数据，用Excel补全日期，再用Vlookupj将已得数据与日期对应到一个sheet中，用R中zoo包中的na.fill（）取相邻数据的均值填充（ts似乎只能生成月度数据，用筛选，查找，替换在excel中确定缺失值）

directory2<-"C:/Users/nsus/Desktop/指数数据全序列(调整汇率R).csv"
#指定文件路径，含字符时要加“”
index_data <-read.csv(directory2,sep=",",header = TRUE)#读入数据
index_data1<-as.data.frame(index_data)#转化为数据框
library(zoo)
## Warning: 程辑包'zoo'是用R版本3.4.3 来建造的
## 
## 载入程辑包：'zoo'
## The following objects are masked from 'package:base':
## 
##     as.Date, as.Date.numeric
index_data1<-as.zoo(index_data)#用zoo包转化数据类型
index_data1<-na.fill(index_data1, "extend") #缺失值扩展均值填充
index_data2<-as.data.frame(index_data1)#转化为数据框
x.Date <- as.Date("2011-08-31") + c(1:2221)#指定数据日期索引
index_data3<-zoo(index_data2,x.Date)#转化数据类型
index_data4<-log(index_data3)#取对数
index_data5<-diff(index_data4)#差分
plot(index_data5,main ="中国、日本、美国、英国、德国国债指数对数收益率时间序列图",xlab="time")#绘制时序图
